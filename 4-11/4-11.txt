clear;
I = imread('Desert.jpg');
I = rgb2gray(I);
J = imnoise(I,'salt & pepper',0.02);
subplot(231),imshow(I);title('origin');
subplot(232),imshow(J);title('salt & pepper');
k1 = medfilt2(J);
k2 = medfilt2(J,[5 5]);
k3 = medfilt2(J,[7 7]);
k4 = medfilt2(J,[9 9]);
subplot(233),imshow(k1);title('3*3中值滤波');
subplot(234),imshow(k2);title('5*5中值滤波');
subplot(235),imshow(k3);title('7*7中值滤波');
subplot(236),imshow(k4);title('9*9中值滤波');
